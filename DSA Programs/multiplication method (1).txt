#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#define MAX 1000
void init();
void store_data();
void display_data();


int htable[MAX];
 int main()
 {
     int choice;
     init();
     while(1)
     {
        printf("\n1.Store data....");
        printf("\n2.Display data....");
        printf("\n3.Exit....");
        printf("\nEnter your choice..");
        scanf("%d",&choice);
        switch(choice)
        {
            case 1:
            store_data();
            break;
            case 2:
            display_data();
            break;
            case 3:
            exit(1);
            default:
            printf("\nWrong choice...");
        }
     }
     return 0;
 }
 void init()
 {
     int i;
     for(i=0;i<MAX;i++)
     {
         htable[i]=-1;
     }
 }
 void store_data()
 {
    int num,hkey,k;
    double a,mul;
    printf("\nEnter a number...");
    scanf("%d",&num);
    printf("\nEnter any constant value between 0 and 1...");
    scanf("%lf",&a);
    k=num;
    mul=k*a;
    hkey=floor(MAX * (mul-floor(mul)));
    htable[hkey]=num;
    printf("\nIndex no = %d and value = %d",hkey,htable[hkey]);
 }
 void display_data()
 {
    int i;
    for(i=0;i<MAX;i++)
    {
        if(htable[i]!=-1)
        printf("\nIndex no = %d and value = %d",i,htable[i]);
    }
 }
