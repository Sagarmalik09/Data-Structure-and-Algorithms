#include<stdio.h>
#include<stdlib.h>
#define MAX 10

int h[MAX];

void init() 
{
	int i;
	for(i=0; i<MAX; i++) 
	{
		h[i] = -1;
	}
}

void insertkeyvalue() 
{
	int key, hk1,hk2,hkey,m0=MAX,m1=m0-1;
	int i,c1=1,c2=2;
	printf("\n Enter the data: ");
	scanf("%d", &key);
	hk1 = key % m0;
	hk2 = key % m1;
	for(i=0;i<MAX;i++) 
	{
		hkey = (hk1+(hk2*i)) % MAX;
		if(h[hkey] == -1) 
		{
			h[hkey] = key;
			break;
		}
	}
}

void display()
{
	int i;
	for(i = 0; i < MAX; i++) 
	{
		if(h[i] != -1)
		printf("\nat index %d \t value = %d", i, h[i]);
	}
}


int main() {
	int choice;
	init();
	while(1) {
		printf("\n 1. Insert.. ");
		printf("\n 2. Display.. ");
		printf("\n 3. Exit.. ");
		printf("\n Enter the choice: ");
		scanf("%d", &choice);
		switch(choice)
		{
			case 1:
				insertkeyvalue();
				break;
			case 2:
				display();
				break;
			case 3:
				exit(1);
			default:
				printf("\n .. Invalid Choice ..\n");
		}
	}
	return 0;
}
