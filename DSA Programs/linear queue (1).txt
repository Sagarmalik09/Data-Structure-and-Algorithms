        {

            res=is_full();

            enqueue();
            break;
            case 2:
            res=is_empty();
            if(res==1)
            dequeue();
            break;

            display();
            break;
            case 4:
            exit(1);
            break;

            printf("Wrong choice...");
            break;
        }
    }
    return 0;
}

{

}
