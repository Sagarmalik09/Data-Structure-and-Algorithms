#include<stdio.h>
#define MAX 100
int sec_larg(int[],int);
int main()
{
    int i,n,res,a[MAX];
    printf("\nEnter the  number of elements..");
    scanf("%d",&n);
    printf("\nEnter %d elements..",n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    res=sec_larg(a,n);
     printf("\nSecond largest number in the array is..%d",res);
     return 0;
}
int sec_larg(int a[],int n)
{
    int i,max=0,secmax=0;
    for(i=0;i<n;i++)
    {
        if(a[i]>max)
        {
            secmax=max;
            max=a[i];
        }
        else if(a[i]>secmax)
        secmax=a[i];
    }
return secmax;
}
