#include<stdio.h>
#include<stdlib.h>
#define MAX 10
int search_no(int[],int,int,int);
int main()
{
    int i,sno,size,arr[MAX],result,beg=0;
    printf("\nEnter the size of array..");
    scanf("%d",&size);
    if((size<=0) || (size>MAX))
    {
        printf("invalid size");
        exit(1);
    }
        printf("\nEnter array elements...\n");
        for(i=0;i<size;i++)
        scanf("%d",&arr[i]);
        printf("\nEnter search number..");
        scanf("%d",&sno);
        result=search_no(arr,beg,size,sno);
        if(result==-1)
        printf("\n%d is not found....,",sno);
        else
        printf("\n%d is found at position %d",sno,result);
        return 0;
}
int search_no(int arr[],int beg,int end,int sno)
{
    int mid;
    if(beg<=end)
    {
    mid=(beg+end)/2;
    if(arr[mid]==sno)
        return mid+1;
    else if(sno<arr[mid])
        return search_no(arr,beg,mid-1,sno);
    else
        return search_no(arr,mid+1,end,sno);
    }
    else
    return -1;
}
