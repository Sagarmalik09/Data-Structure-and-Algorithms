void abs_diff(int arr[],int size)
{
    int i,key,min,res,n;
    printf("\nEnter the key element..");
    scanf("%d",&key);
    printf("\nAbsolute diffrences are..");
    for(i=0;i<size;i++)
    {
        res=abs(arr[i]-key);
        printf("\n%d - %d = %d",arr[i],key,res);
        if(res<min)
        {
            min=res;
            n=arr[i];
        }
    }
    printf("\nMinimum absolute difference is %d of value %d",min,n);
}
