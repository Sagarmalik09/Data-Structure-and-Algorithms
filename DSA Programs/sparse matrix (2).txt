//check tri-diagonal matrix

#include<conio.h>
#include <stdio.h>

int main()
{
    int sparse[5][5]={
                        {1,1,0,0,0},
                        {5,2,8,0,0},
                        {0,8,3,2,0},
                        {0,0,4,1,5},
                        {0,0,0,7,9}
    };
    int i,j,check=1;
    for(i=0;i<5;i++)
    {
        for(j=0;j<5;j++)
        {
            printf("\t%d",sparse[i][j]);
        }
        printf("\n");
    }
    for(i=0;i<5;i++)
    {
        for(j=0;j<5;j++)
        {
            if(sparse[i][j]!=0 && (i==j || i-1==j || i==j-1))
            {
                if(sparse[i][j]==0)
                {
                    check=0;
               }
            }
            else if(sparse[i][j]!=0)
            {
                check=0;
            }
        }
    }
    if(check==1)
    printf("\nTri-diagonal matrix");
    else
    printf("\nNot a tri-diagonal matrix");
    return 0;
}

//check upper-triangular matrix

#include<conio.h>
#include <stdio.h>

int main()
{
    int sparse[5][5]={
                        {1,1,2,5,8},
                        {0,2,8,9,7},
                        {0,0,3,7,2},
                        {0,0,0,1,5},
                        {0,0,0,0,9}
    };
    int i,j,check=0;
    for(i=0;i<5;i++)
    {
        for(j=0;j<5;j++)
        {
            printf("\t%d",sparse[i][j]);
        }
        printf("\n");
    }
    for(i=0;i<5;i++)
    {
        for(j=0;j<5;j++)
        {
            if(sparse[i][j]==0 && i>j)
                check=1;
            else if(sparse[i][j]!=0 && i>j)
            {
                check=0;
                break;
            }
        }
        if(check==0 && i>0)
        break;
    }
    if(check==1)
    printf("\nUpper-triangular matrix");
    else
    printf("\nNot an upper-triangular matrix");
    return 0;
}
