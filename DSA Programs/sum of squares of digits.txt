
#include <stdio.h>

int main()
{
   int i,j,row,column,arr[10][10],flag=0;
   printf("\nEnter the number of rows and columns..");
   scanf("%d%d",&row,&column);
   printf("\nEnter the elements..");
   for(i=0;i<row;i++)
   {
       for(j=0;j<column;j++)
       {
           scanf("%d",&arr[i][j]);
       }
   }
    for(i=0;i<row;i++)
   {
       for(j=0;j<column;j++)
       {
           if((i==j) && (arr[i][j]!=1))
            {
                flag=-1;
                break;
            }
            if((i!=j) && (arr[i][j]!=0))
            {
                flag=-1;
                break;
            }
       }
   }
   if(flag==0)
   printf("\nIdentity matrix..");
   else
   printf("\nNot an Identity matrix..");
    return 0;
}
